1. download and start redis
docker pull redis 
docker run -p 6379:6379 --name redis_instance -t redis

2. download and start phore db updater
clone this repo:
git clone https://github.com/phoreproject/Phore-RPC-Proxy
cd ./Phore-RPC-Proxy/script/keepPhoreUpdated
docker build -t keep_phore_updated .

for sharing docker volume on host:
mkdir phore_database_backup
docker run -t keep_phore_updated -v phore_database_backup:/data



3. Set environmental variables
REDIS_HOST = redis docker ip
REDIS_PORT = 6379 (default)


4. Start phored instance
.\phored.exe -server -printtoconsole -rest -blocknotify="node ./blocknotify.js blocknotify %s"

or for transactions: 
-mempoolnotify="node ./blocknotify.js mempoolnotify %s"

5. Download node packages
npm install in project directory.

5. Start index.js
node index.js
